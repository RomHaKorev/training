
\newcommand\twocolumns[3]{
  \FPeval{\leftsize}{#1}
  \FPeval{\rightsize}{1 - #1}

  \begin{columns}
    \begin{column}{\leftsize\textwidth}
      #2
    \end{column}
    \begin{column}{\rightsize\textwidth}
      #3
    \end{column}
  \end{columns}
}
