\begin{frame}{Clean Code principles}
  \begin{itemize}
  \item KISS: Keep It Simple, Stupid
  \item DRY: Don't Repeat Yourself
  \item YAGNI: You Aren't Gonna Need It
  \item Favor readability over conciseness: A code is always obvious for the machine
  \end{itemize}
\end{frame}


\begin{frame}{Clean Code principles}
  \begin{center}
  KISS: Keep It Simple, Stupid
  \end{center}
  \begin{itemize}
    \item Be humble: your code has to be simple for everyone
    \item Break down problems into many small problems
    \item Each method should only solve one little problem, not many uses cases
    \item Solve the problem, then code it
  \end{itemize}
\end{frame}


\begin{frame}{Clean Code principles}
  \begin{center}
  DRY: Don't Repeat Yourself
  \end{center}
  \begin{itemize}
    \item Remove code redundancy
    \item Don't repeat the knowledge: Work on the patterns not the code
    \item{Rule Of Three}
    \begin{itemize}
      \item First time you have to do something, just do it
      \item Second time you have to do the same thing, do it again
      \item Third time you have to do it, start refactoring
    \end{itemize}
  \end{itemize}
\end{frame}



\begin{frame}{Clean Code principles}
  \begin{center}
  YAGNI: You Aren't Gonna Need It
  \end{center}
  \begin{itemize}
    \item Do the simplest thing that could possibly work
    \item Don't anticipate. Open your code
  \end{itemize}
\end{frame}

\begin{frame}{Bad reasons to refactor the code}
  \begin{itemize}
    \item New technology is not a sufficient reason
    \item If you don't need it, don't touch it
    \item Ego and Personal preferences don't matter
  \end{itemize}
