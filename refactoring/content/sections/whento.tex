\section{When to refactor}


%## Before you refactor
%## Continuous refactor
%### During a code review
%### Before adding a feature or fixing a bug


\begin{frame}{Before your refactor}
  \begin{itemize}
  	\item Avoid the temptation to rewrite everything
  	\item Incremental changes. No massive changes
  	\item The tests pass after each iteration
  	\item Be rational: no personal preferences or ego
  	\item A new technology is not a good reason to refactor
  \end{itemize}
\end{frame}

\begin{frame}{During a code review}
  \begin{itemize}
  	\item Without an author: the code is obvious
  	\item With an author: the code is logic
  \end{itemize}
\end{frame}

\begin{frame}{Before adding a feature or fixing a bug}
  \begin{itemize}
  \item Refactoring helps you to understand the code
  \item In a clean code, bugs are obvious
  \item Much easier to insert a new feature in a clean code 
  \end{itemize}
\end{frame}
